 子集

给定一组**不含重复元素**的整数数组 *nums*，返回该数组所有可能的子集（幂集）。

**说明：**解集不能包含重复的子集。

**示例:**

```
输入: nums = [1,2,3]
输出:
[
  [3],
  [1],
  [2],
  [1,2,3],
  [1,3],
  [2,3],
  [1,2],
  []
]
```

```c++
//递归
//https://www.youtube.com/watch?v=3dEVYiyFKac
//时间复杂度o(n*2^n),空间复杂度(o(n*2^n))
class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<int> sub;
        vector<vector<int>> res;
        subsets(nums,0,sub,res);
        return res;
    }
public:
    void subsets(vector<int>& nums,int i,vector<int>& sub,vector<vector<int>>& res){
        if(i==nums.size()){
            res.push_back(sub);
            return;
        }
        sub.push_back(nums[i]);
        subsets(nums,i+1,sub,res);
        sub.pop_back();
        subsets(nums,i+1,sub,res);
    }
};
//迭代
class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> res={{}};
        for(int i=0;i<nums.size();i++){
            int n=res.size();
            for(int j=0;j<n;j++){
                res.push_back(res[j]);
                res.back().push_back(nums[i]);
            }
        }
        return res;
    }
};
```
