请你写出一个能够举单词全部缩写的函数。

注意：输出的顺序并不重要。

示例：

输入: "word"
输出:
["word", "1ord", "w1rd", "wo1d", "wor1", "2rd", "w2d", "wo2", "1o1d", "1or1", "w1r1", "1o2", "2r1", "3d", "w3", "4"]

```c++
//https://www.cnblogs.com/grandyang/p/5261569.html
/*
0000 word
0001 wor1
0010 wo1d
0011 wo2
0100 w1rd
0101 w1r1
0110 w2d
0111 w3
1000 1ord
1001 1or1
1010 1o1d
1011 1o2
1100 2rd
1101 2r1
1110 3d
1111 4
*/
//位操作法
//凡是0的地方都是原来的字母，单独的1还是1，如果是若干个1连在一起的话，就要求出1的个数，用这个数字来替换对应的字母
class Solution {
public:
    vector<string> generateAbbreviations(string word) {
        vector<string> res;
        for(int i=0;i<pow(2,word.size());i++){
            int cnt = 0, t=i;
            string tmp="";
            for(int j=0;j<word.size();j++){
                if(t&1==1){
                    cnt++;
                    if(j == word.size()-1){
                        tmp += to_string(cnt);
                    }
                }else{
                    if(cnt>0){
                        tmp += to_string(cnt);
                        cnt = 0;
                    }
                    tmp += word[j];
                }
                t >>= 1;
            }
            res.push_back(tmp);
        }
        return res;
    }
};
//回溯法
class Solution {
public:
    vector<string> generateAbbreviations(string word) {
        string tmp="";
        vector<string> res;
        helper(word,res,tmp,0,0);
        return res;
    }
private:
    // i is the current position
    // k is the count of consecutive abbreviated characters
    void helper(string& word, vector<string>& res, string tmp,int i,int k){
        if(i==word.size()){
            if(k>0){
                tmp += to_string(k);
            }
            res.push_back(tmp);
        }else{
            helper(word,res,tmp,i+1,k+1);
            if(k>0){
                tmp += to_string(k); 
            }
            tmp += word[i];
            helper(word,res,tmp,i+1,0);
        }
    }
};
```
