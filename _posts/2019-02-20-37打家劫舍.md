打家劫舍

你是一个专业的小偷，计划偷窃沿街的房屋。每间房内都藏有一定的现金，影响你偷窃的唯一制约因素就是相邻的房屋装有相互连通的防盗系统，**如果两间相邻的房屋在同一晚上被小偷闯入，系统会自动报警**。

给定一个代表每个房屋存放金额的非负整数数组，计算你**在不触动警报装置的情况下，**能够偷窃到的最高金额。

**示例 1:**

```
输入: [1,2,3,1]
输出: 4
解释: 偷窃 1 号房屋 (金额 = 1) ，然后偷窃 3 号房屋 (金额 = 3)。
     偷窃到的最高金额 = 1 + 3 = 4 。
```

**示例 2:**

```
输入: [2,7,9,3,1]
输出: 12
解释: 偷窃 1 号房屋 (金额 = 2), 偷窃 3 号房屋 (金额 = 9)，接着偷窃 5 号房屋 (金额 = 1)。
     偷窃到的最高金额 = 2 + 9 + 1 = 12 。
```

//step
//1 Find recursive relation
//2 Recursive (top-down)
//3 Recursive + memo (top-down)
//4 Iterative + memo (bottom-up)
//5 Iterative + N variables (bottom-up)

// recursive relation
//rob(i) = Math.max( rob(i - 2) + currentHouseValue, rob(i - 1) )
//Run Time limit 
class Solution {
public:
    int rob(vector<int>& nums) {
         int n = nums.size();
         return rob(nums,n-1);
    }
private:
	int rob(vector<int>& nums, int n){
		 if(n<0){
		 	return 0;
		 }
		 return max(rob(nums,n-2)+nums[n],rob(nums,n-1));
	}
};
//Recursive + memo (top-down).12ms
class Solution {
public:
    int rob(vector<int>& nums) {
         int n = nums.size();
	     vector<int> mem;
	     for(int i=0;i<n;i++){
	     	mem.push_back(-1);
	     }
         return rob(nums,n-1,mem);
    }
private:
	int rob(vector<int>& nums, int n,vector<int>& mem){
		 if(n<0){
		 	return 0;
		 }
		 if(mem[n]>=0){
		 	return mem[n];
		 }
		 int result = max(rob(nums,n-2,mem)+nums[n],rob(nums,n-1,mem));
		 mem[n]=result;
		 return result;
	}
};
// Iterative + memo (bottom-up) 8 ms
class Solution {
public:
    int rob(vector<int>& nums) {
         int n = nums.size();
         if(n==0){
         	return 0;
         }
	     vector<int> mem;
	     for(int i=0;i<n+1;i++){
	     	mem.push_back(-1);
	     }
	     mem[0]=0;
	     mem[1]=nums[0];
	     for(int i=1;i<n;i++){
	     	mem[i+1]=max(mem[i],mem[i-1]+nums[i]);
	     }
	     return mem[n];
	}
};
//Iterative + 2 variables (bottom-up) (best algs)
class Solution {
public:
    int rob(vector<int>& nums) {
        int n = nums.size();
        int pre1 = 0;
        int pre2 = 0;
        for(int i=0;i<n;i++){
        	int tmp = pre1;
        	pre1 = max(pre2+nums[i],pre1);
        	pre2 = tmp;
        }
        return pre1;
    }
};
