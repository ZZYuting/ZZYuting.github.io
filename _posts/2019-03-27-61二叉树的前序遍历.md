给定一个二叉树，返回它的 *前序* 遍历。

 **示例:**

```
输入: [1,null,2,3]  
   1
    \
     2
    /
   3 

输出: [1,2,3]
```

**进阶:** 递归算法很简单，你可以通过迭代算法完成吗？

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
//recursion
class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> res;
        preorder(root,res);
        return res;
    }
private:
    void preorder(TreeNode* root,vector<int>& res){
        if(root==NULL){
            return;
        }
        res.push_back(root->val);
        if(root->left){
            preorder(root->left,res);
        }
        if(root->right){
            preorder(root->right,res);
        }
    }
};

//iterative
class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> res;
        if(root==NULL)
            return res;
        stack<TreeNode*> stack;
        stack.push(root);
        while(!stack.empty()){
            TreeNode* tmp=stack.top();
            res.push_back(tmp->val);
            stack.pop();
            if(tmp->right!=NULL){
                stack.push(tmp->right);
            }
            if(tmp->left!=NULL){
                stack.push(tmp->left);
            }
        }
        return res;
    }
};
```
