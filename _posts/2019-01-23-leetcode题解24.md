将两个有序链表合并为一个新的有序链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。 

**示例：**

```c++
输入：1->2->4, 1->3->4
输出：1->1->2->3->4->4
```

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
//迭代的方法
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode* dummy=new ListNode(0);
        ListNode* tmp=dummy;
        while(l1!=nullptr&&l2!=nullptr){
            if(l1->val<l2->val){
                tmp->next=l1;
                l1=l1->next;
            }else{
                tmp->next=l2;
                l2=l2->next;
            }
            tmp=tmp->next;
        }
        if(l1!=nullptr){
            tmp->next=l1;
        }
        if(l2!=nullptr){
            tmp->next=l2;
        }
        return dummy->next;        
    }
};
```
