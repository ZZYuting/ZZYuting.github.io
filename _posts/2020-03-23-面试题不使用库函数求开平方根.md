```c++
//不使用库函数求开平方根
class Solution {
public:
    double SQRT(double n)
    {
        if(n<=1) return n;
        double l = 1, r = n;
        while(r - l > 10e-6)
        {
            double mid = (r + l) / 2;
            if(mid * mid > n)
                r = mid;
            else
                l = mid;
        }
        return (l + r) / 2;
    }
};
```

```
 x 的平方根
实现 int sqrt(int x) 函数。

计算并返回 x 的平方根，其中 x 是非负整数。

由于返回类型是整数，结果只保留整数的部分，小数部分将被舍去。

示例 1:

输入: 4
输出: 2
示例 2:

输入: 8
输出: 2
说明: 8 的平方根是 2.82842..., 
     由于返回类型是整数，小数部分将被舍去。
```

```c++
class Solution {
public:
    int mySqrt(int x) {
        if(x <= 1) return x;
        int l = 1, r = x;
        while(l<r){
            int mid = l + (r-l)/2;
            if(mid <= x/mid && (mid+1) > x/(mid+1)){
                return mid;
            }else if(mid > x/mid){
                r = mid-1;
            }else{
                l = mid + 1;
            }
        }
        return l;
    }
};
```
