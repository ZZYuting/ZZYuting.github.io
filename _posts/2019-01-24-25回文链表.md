请判断一个链表是否为回文链表。

**示例 1:**

```
输入: 1->2
输出: false
```

**示例 2:**

```
输入: 1->2->2->1
输出: true
```

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
         ListNode *fast=head,*cur=head,*pre=NULL,*next=NULL;
         while(fast!=NULL&&fast->next!=NULL){
         	fast=fast->next->next;
         	next=cur->next;
         	cur->next=pre;
         	pre=cur;
         	cur=next;
         }
         if(fast!=NULL){
         	cur=cur->next;
         }
         while(pre!=NULL){
         	if(pre->val!=cur->val){
         		return false;
         	}
         	else{
         		pre=pre->next;
         		cur=cur->next;
         	}
         }
         return true;
    }
};
